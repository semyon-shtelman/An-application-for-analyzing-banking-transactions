54def get_cards_info(df: pd.DataFrame) -> list[dict]:

56 logger.info("Начало расчета информации по картам")

59 logger.warning("DataFrame пуст для расчета информации по картам")

66 return [] 

67 

68 

69 try: 

70 cards_info = [] 

71 

72 for card_number, group in transactions.groupby("Номер карты"): 

73 total_spent = abs(group["Сумма платежа"].sum()) 

74 total_spent_rounded = round(total_spent, 2) 

75 cashback = round(total_spent / 100, 2) 

76 

77 cards_info.append({ 

78 "last_digits": card_number, 

79 "total_spent": total_spent_rounded, 

80 "cashback": cashback

82 logger.info(f"Рассчитана информация по {len(cards_info)} картам") 

83 return cards_info 

84 except Exception as e: 

85 logger.error(f"Ошибка при расчете информации по картам: {e}")

136def get_filtering_df_by_date(df: pd.DataFrame, target_date: str) -> pd.DataFrame: 

137 """Фильтрует данные с начала месяца, на который выпадает входящая дата, по входящую дату.""" 

138 logger.info(f"Фильтрация данных по дате: {target_date}") 

139 

140 if df.empty: 

141 logger.warning("DataFrame пуст для фильтрации по дате") 

142 return df 

143 

144 try: 

145 date_dt = pd.to_datetime(target_date, format="%Y-%m-%d %H:%M:%S") 

146 start_of_month = date_dt.replace(day=1) 

147 

148 df_dates = pd.to_datetime(df["Дата платежа"], format="%d.%m.%Y") 

149 filtered_df = df[df_dates.between(start_of_month, date_dt)] 

150 logger.info(f"Данные отфильтрованы. Записей до: {len(df)}, после: {len(filtered_df)}") 

151 return filtered_df 

152 except Exception as e: 

153 logger.error(f"Ошибка при фильтрации данных по дате: {e}") 

154 return pd.DataFrame()

157def get_currency_rates(currencies: list[str]) -> list: 

158 """Получает курсы валют с ЦБ РФ""" 

159 logger.info(f"Запрос курсов валют: {currencies}") 

160 

161 url = "https://www.cbr-xml-daily.ru/daily_json.js" 

162 currency_rates = [] 

163 

164 if not currencies: 

165 logger.info("Список валют пуст") 

166 return [] 

167 

168 for currency in currencies: 

169 try: 

170 logger.debug(f"Запрос курса для валюты: {currency}") 

171 response = requests.get(url, timeout=10) 

172 response.raise_for_status() 

173 data = response.json() 

174 

175 rate = data.get("Valute", {}).get(currency, {}).get("Value") 

176 if rate: 

177 currency_rates.append({"currency": currency, "rate": round(float(rate), 2)}) 

178 logger.debug(f"Получен курс для {currency}: {rate}") 

179 else: 

180 logger.warning(f"Курс для валюты {currency} не найден в ответе") 

181 

182 except requests.exceptions.RequestException as e: 

183 logger.error(f"Ошибка запроса курса валюты {currency}: {e}") 

184 continue 

185 except Exception as e: 

186 logger.error(f"Неожиданная ошибка при получении курса {currency}: {e}") 

187 continue 

188 

189 logger.info(f"Получены курсы для {len(currency_rates)} валют из {len(currencies)} запрошенных") 

190 return currency_rates

194 """Получает цены акций из S&P500""" 

195 logger.info(f"Запрос цен акций: {symbols}")

202 logger.info("Список акций пуст")

216 logger.debug(f"Получена цена для {symbol}: {price}") 

217 else: 

218 logger.warning(f"Цена для акции {symbol} не найдена в ответе") 

219 

220 except requests.exceptions.RequestException as e: 

221 logger.error(f"Ошибка запроса цены акции {symbol}: {e}") 

222 continue 

223 except Exception as e: 

224 logger.error(f"Неожиданная ошибка при получении цены акции {symbol}: {e}") 

225 continue 

226
